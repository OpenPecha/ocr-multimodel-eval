སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སུམངྒལ། སརྦ་བྷ་ཡ་ སྭ་སྟི་ སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། སྣ་ཚོགས་སེམས་ཅན་གྱི་རྒྱུད་བསྙེལ་བའི་ཡེ་ཤེས་དགྱེས་བཞིན་དུ་འཕྲོ་བའི་རྒྱུད། ། །འོན་ཀྱང་དབང་དུ་འདུ་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་བསྒྲུབ་པར་བྱེད་པ་སྟེ། །བསླབ་པ་དང་བསྟན་བཅོས་སུ་བསླབ་པས་རང་གི་དབང་དུ་བསྒྱུར་བ་སྒྲུབ་པར་བྱེད་ལ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། །འཇིག་རྟེན་པ་ཡང་དབང་བསྒྱུར་བའི་གོ་འཕང་དུ་འགྲོ་བའི་ཚུལ་གྱིས་སྟོན་པ་སྟེ། །དེ་ལ་ཡང་དབང་དུ་འདུ་བར་བྱེད་པ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བས་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སེམས་ཅན་རྣམས་ཀྱང་དབང་བསྒྱུར་བ་ལ་དགའ་སྟེ། །དེ་ལ་དབང་བསྒྱུར་བ་ནི་རང་གི་དབང་དུ་བསྒྱུར་བ་དང་དེ་ལ་ཡང་དབང་བསྒྱུར་བར་བྱེད་པ་རྣམས་ཀྱང་དེ་ལྟར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ཡང་སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲུབ་པར་གསུངས་སོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡོན་ཏན་ལས་ཀྱང་དམ་པ་གཞན་དུ་འགྲོ་བའི་རྣམ་པར་སྟོན་པ་དེ་ལས་ཀྱང་ཡོན་ཏན་ལྷག་པ་དེ་དོན་དུ་གཉེར་བ་སྟེ། །སྔར་གྱི་གདམ་པ་རྣམས་ཀྱང་དེ་ལྟར་འགྲོ་བར་གསུངས་སོ། །ཡང་མདོ་ལས། ཨཱརྱ་ཛམ་བྷ་ལཻ་མནྟྲ་ཛཻ་ཝ་རི་སཻ་སཻ་སཻ་མ་སཻ་མཱ་ཧཱ་ཛམ་བྷ་ལཻ་དྷ་རི་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་བྷ་རི་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་དྷ་རི་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་མཱ་ཏྲ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཧཱ་སཻ་སཻ་མཱ་ཏྲ་
